139      def decode_model(self, raw_model): 
140          try: 
141              decoded_model = [] 
142              try: 
143                  for layer in raw_model: 
144                      single_layer = [layer[0], 'hidden', layer[1].get(), layer[2].get()] 
145                      decoded_model.append(single_layer) 
146                      # print("ID=" + str(layer[0]) + ", Neurons=" + str(layer[1].get()) + ", Activation=" + str(layer[2].get())) 
147                  output_layer = decoded_model.pop() 
148              except Exception: 
149                  return None 
150              output_layer[1] = 'output' 
151              output_layer[2] = self.output_classes 
152              decoded_model.append(output_layer) 
153              return decoded_model 
154          except Exception: 
155              return None

157      def model_network(self, algorithm='LevenbergMarquardt', model=None): 
158   
159          model = self.decode_model(model)

161              model = [

180              if layer[3] == 'Tanh': 
181                  layer_model.append(layers.Tanh(layer[2])) 
182              if layer[3] == 'Softplus': 
183                  layer_model.append(layers.Softplus(layer[2])) 
184              if layer[3] == 'Softmax': 
185                  layer_model.append(layers.Softmax(layer[2])) 
186              if layer[3] == 'Elu': 
187                  layer_model.append(layers.Elu(layer[2]))
